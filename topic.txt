# команда для сохранения топика в консоли контейнера kafka1
/bin/kafka-topics --create --topic notification-topic --bootstrap-server kafka1:29092 --partitions 3 --replication-factor 2
# ну или для sh
/bin/kafka-topics.sh --create --topic notification-topic --bootstrap-server kafka1:29092 --partitions 3 --replication-factor 2
